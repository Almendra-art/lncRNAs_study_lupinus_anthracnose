#remove_rfam.sh
#!/bin/bash

awk '
!/^#/ {
    family=$2

    if (family ~ /^SSU_rRNA/ ||
        family ~ /^LSU_rRNA/ ||
        family == "5_8S_rRNA" ||
        family == "tRNA" ||
        family == "U2" ||
        family == "U3" ||
        family == "U6" ||
        family == "Plant_U3" ||
        family ~ /^snoR/ ||
        family ~ /^snoZ/ ||
        family ~ /^SNOR/ ||
        family ~ /_SRP$/ )
    {
        print $4
    }
}
' rfam.tblout | sort -u > remove_lncRNAs_rfam.txt

#UTR filtering: 

#location of transcripts lncRNAs:
grep -F -f final_lncRNA.txt stringtie_merged.gtf > lncRNAs_location.gtf
awk '$3=="transcript"' lncRNAs_location.gtf > lncRNAs_location_transcripts.gtf 

#Create bed file 
awk '$3=="transcript"{
    match($0, /transcript_id "([^"]+)"/, a);
    print $1"\t"($4-1)"\t"$5"\t"a[1]"\t.\t"$7
}' lncRNAs_location_transcripts.gtf > lncRNAs.bed

#location of UTR regions 
grep -P "\tfive_prime_UTR\t|\tthree_prime_UTR\t" Lupinus_luteus.Genome.V2.gff3 > UTRs.gff3

#create bed file
awk 'BEGIN{OFS="\t"} $3=="five_prime_UTR" || $3=="three_prime_UTR"{
print $1,$4-1,$5,$3,".",$7
}' utrs.gff3 > utrs.bed
